#!/usr/bin/python3

class Solution:
    def canPlaceFlowers(self, flowerbed: List[int], n: int) -> bool:
        if(n==0):
            return True
        dum=flowerbed[:]
        res_bool=[]
        count=0
        for i in range(0,len(flowerbed)):
            if(flowerbed[i]==0  and (i+1==len(flowerbed) or flowerbed[i+1]==0) and (i==0 or flowerbed[i-1]==0)):
                flowerbed[i]=1
                n-=1
                if(n==0):
                    return True
        return False
